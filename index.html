        // Script simples para carregar imagens locais na prévia

        document.getElementById('imageInput').addEventListener('change', function(e) {

            const files = e.target.files;

           

            // Mapa dos nomes de arquivo para os IDs das imagens no HTML

            // Adapte estes nomes se os seus arquivos tiverem nomes diferentes!

            const imageMap = {

                'image_824de1.png': 'img-logo',

                'image_81f744.png': 'img-nivel1',

                'image_81f788.png': 'img-nivel2',

                'image_81f43c.png': 'img-nivel3'

            };

 

            for (let i = 0; i < files.length; i++) {

                const file = files[i];

                // Verifica se o nome do arquivo existe no nosso mapa

                // (Tenta combinar o nome exato ou parte dele caso o navegador renomeie)

                let targetId = imageMap[file.name];

               

                // Se não achar pelo nome exato, tenta achar pelo ID diretamente se você renomeou os arquivos para nivel1.png, etc.

                if (!targetId) {

                   // Fallback genérico: se o usuário carregar 4 imagens, tenta adivinhar

                   console.log("Arquivo carregado: " + file.name);

                }

 

                if (targetId) {

                    const imgElement = document.getElementById(targetId);

                    if (imgElement) {

                        const reader = new FileReader();

                        reader.onload = function(e) {

                            imgElement.src = e.target.result;

                            // Remove o estilo de erro se tiver

                            imgElement.style.padding = '0';

                            imgElement.style.border = 'none';

                        };

                        reader.readAsDataURL(file);

                    }

                }

            }
